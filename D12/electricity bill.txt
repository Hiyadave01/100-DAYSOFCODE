Q24: Write a program to calculate an electricity bill based on units consumed.

Sample Test Cases:
Input 1:
50
Output 1:
Bill: ₹250

Input 2:
150
Output 2:
Bill: ₹950

Input 3:
250
Output 3:
Bill: ₹2200


#include<stdio.h>
int main() {
    int unitsconsumed;
    
    printf("Enter the number of units consumed :");
    scanf("%d", &unitsconsumed);

    if(unitsconsumed >=1 && unitsconsumed <=100) {
        printf("The rate will be Rs5/unit.\n");
    }
    else if(unitsconsumed >=101 && unitsconsumed <= 200) {
        printf("The rate will be Rs7/unit.\n");
    }
    else if(unitsconsumed >= 201 && unitsconsumed <= 300) {
        printf("The rate will be Rs10/unit.\n");
    }
    else if(unitsconsumed > 300) {
        printf("The rate will be Rs12/unit.\n");
    }
    else {
       printf("Invalid input!\n");
    }
   return 0; 
}
