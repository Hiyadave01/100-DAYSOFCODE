Q23: Write a program to calculate a library fine based on late days.

Sample Test Cases:
Input 1:
4
Output 1:
Fine ₹8

Input 2:
8
Output 2:
Fine ₹32

Input 3:
15
Output 3:
Fine ₹90

Input 4:
31
Output 4:
Membership Cancelled

#include<stdio.h>
int main()
 {
    int daysLATE, fine;

    //Input the number of late days
    printf("Enter number of late days:");
    scanf("%d", &daysLATE);

    //Calculate fine 
    if(daysLATE >=1 && daysLATE <=5){
        printf("Fine : Rs2/day\n");
    }
    else if(daysLATE >=6 && daysLATE <=10){
        printf("Fine : Rs4/day\n");
    }
    else if(daysLATE >=11 && daysLATE <=30){
        printf("Fine : Rs6/day\n");
    }
    else if(daysLATE >30){
        printf("Membership Cancelled!\n");
    }
    else {
        printf("Invalid input.\n");
    }
}
