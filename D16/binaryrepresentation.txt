Q31: Write a program to take a number as input and print its equivalent binary representation.

Sample Test Cases:
Input 1:
10
Output 1:
1010

Input 2:
7
Output 2:
111


#include <stdio.h>
int n, rem, i = 1;
long binary = 0;
int main()
{

    printf("Enter a decimal number: ");
    scanf("%d", &n);

    if (n < 0)
    {
        printf("Invalid Input");
        return 1;
    }
    else if (n == 0)
    {
        printf("Binary : 0\n");
        return 0;
    }
    while (n != 0)
    {
        rem = n % 2;                             // values in 0 or 1
        binary = binary + rem * i;         // i is the place value initial
        n = n / 2;
        i = i * 10; // place value is multiplied by 10
    }
    printf("Binary : %1ld\n", binary);
    return 0;
}
