Q27: Write a program to print the sum of the first n odd numbers.

Sample Test Cases:
Input 1:
3
Output 1:
9

Input 2:
5
Output 2:
25

#include<stdio.h>

int main()
{
    int n, i = 1, sum = 0, odd = 1;

    printf("Enter the value of n: ");
    scanf("%d", &n);

    //Loop to calculate the sum of n odd numbers
    for(i = 1; i <= n; i++)
    {
        sum = sum + odd;
        printf("%d\n", odd);
        odd = odd + 2; //Incrementing the odd number by 2
    }

    printf("The sum of first %d odd numbers is: %d\n", n, sum);
    return 0;
}
