Q28: Write a program to print the product of even numbers from 1 to n.

Sample Test Cases:
Input 1:
4
Output 1:
8 (2 * 4)

Input 2:
6
Output 2:
48 (2 * 4 * 6)


#include<stdio.h>
int main()
{
    int n, i = 1, sum = 0, even = 2;
    printf("Enter the value of n:");
    scanf("%d", &n);

    //Loop to calculate the sum of n even numbers
    for(i = 1; i <= n; i++)
    {
        sum += even;
        printf("The %d even number is: %d\n", i, even);
        even += 2; //Incrementing the even number by 2
    }
    printf("The sum of first %d even numbers is: %d\n", n, sum);
    return 0;
}
