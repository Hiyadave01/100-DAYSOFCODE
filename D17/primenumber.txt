Q34: Write a program to check if a number is prime.

Sample Test Cases:
Input 1:
7
Output 1:
Prime

Input 2:
10
Output 2:
Not prime


#include<stdio.H>
int main()
{
    int n , i , Prime = 2;
    printf("Enter an integer: ");
    scanf("%d", &n);

    //Prime no is greater than 1
    if (n <= 1)
      {
        printf("No prime number exists below 2.");
        return 0;
      }

    //Finding factors from 2 to n/2
    for(i = 2; i <= n/2; i++)
      if(n % i == 0)
        {
          Prime = 0;
          break;
        }
    if (Prime)
      {
        printf("%d is a prime number.", n);
      }
    else
    {
      printf("%d is not a prime number.", n);
    }
    return 0;
    
}
