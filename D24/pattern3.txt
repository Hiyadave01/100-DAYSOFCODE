Q48: Write a program to print the following pattern:
1
12
123
1234
12345

/*
Sample Test Cases:
Input 1:

Output 1:
1
12
123
1234
12345

*/

#include <stdio.h>
int main(){
    int p, q, i, j;
    printf("Enter the number of rows and columns: ");
    scanf("%d %d", &p, &q);

    for ( i = 1; i <= p; i++) {
        for ( j = 1; j <= i; j++) {
            printf("%d", j);
        }
        printf("\n");
    }
    return 0;
}
