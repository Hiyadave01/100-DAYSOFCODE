Q47: Write a program to print the following pattern:
*
**
***
****
*****

/*
Sample Test Cases:
Input 1:

Output 1:
*
**
***
****
*****

*/

#include <stdio.h>
int main() {
    int a, b, i, j;

    printf("Enter the number of rows and columns: ");
    scanf("%d %d", &a, &b);
    
    for (int i = 1; i <= a; i++) {
        for (int j = 1; j <= i; j++) {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}
