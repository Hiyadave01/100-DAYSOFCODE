#include<stdio.h>

int main() {
    int celsius, fahrenheit;

    //Input temperature in Celsius
    printf("Enter temperature in Celsius: ");
    scanf("%d", &celsius);

    //Convert Celsius to Fahrenheit
    fahrenheit = (celsius * 9/5) + 32;
    printf("Temperature in Fahrenheit: %d\n", fahrenheit);
    return 0;
    
}
