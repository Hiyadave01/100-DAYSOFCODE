Q18: Write a program to assign grades based on a percentage input.

Sample Test Cases:
Input 1:
95
Output 1:
Grade A

Input 2:
82
Output 2:
Grade B

Input 3:
68
Output 3:
Grade D

Input 4:
50
Output 4:
Grade F

#include<stdio.h>

int main(){
    float marks;
    printf("Enter your marks:");
    scanf("%f", &marks);
    if(marks>=90)
        { 
           printf("Your grade is A.\n");
        }
    else if(marks>=80)
        {
            printf("Your grade is B.\n");
        }
    else if(marks>=70)
        {
            printf("Your grade is C.\n");
        }
    else if(marks>=60)
        {
            printf("Your grade is D.\n");
        }
    else
        {
            printf("Your grade is F.\n");
        }
        return 0;
    
}
