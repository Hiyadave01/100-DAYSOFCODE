Q46: Write a program to print the following pattern:
*****
*****
*****
*****
*****

/*
Sample Test Cases:
Input 1:

Output 1:
*****
*****
*****
*****
*****

*/


#include <stdio.h>
int main() {
    int p, q, i, j;

    printf("Enter the number of rows and columns: ");
    scanf("%d %d", &p, &q);
    for (int i = 1; i <= p; i++) {
        for (int j = 1; j <= q; j++) {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}
